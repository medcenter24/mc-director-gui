To use it we need module MultiSelectorModule

Cities

{{ 'Selected' | translate }}: {{ city.title }}; 

              
                <nga-multi-selector
                  [service]="cityService"
                  placeholder="Cities"
                  labelField="title"
                  (init)="startLoader($event)"
                  (loaded)="stopLoader($event)"
                  (selected)="cities = $event"
                ></nga-multi-selector>
